CREATE trigger silme on Ogrenci
after delete
as
begin
   if(exists(select * from deleted where cinsiyet='Kýz' and Sube='A'))
    begin
     raiserror('A subesindeki kýz öðrencileri silemezsiniz!!',0,0)
     rollback transaction
    end
end
